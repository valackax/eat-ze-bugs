
def sum_and_product_of_digits(d):

    # Виділяємо кожну цифру чотиризначного числа
    thousands = d // 1000          # Отримуємо тисячі (1)
    hundreds = (d // 100) % 10     # Отримуємо сотні (3)
    tens = (d // 10) % 10          # Отримуємо десятки (5)
    ones = d % 10                  # Отримуємо одиниці (2)
    # Обчислюємо суму (S) та добуток (D)
    S = thousands + hundreds + tens + ones
    D = thousands * hundreds * tens * ones

    return S, D

# Вхідні дані
d = 1352

# Виклик функції
S, D = sum_and_product_of_digits(d)

# Виведення результату
print(f"Вхідні дані: d = {d}")
print(f"Вихідні дані: S = {S}, D = {D}")
