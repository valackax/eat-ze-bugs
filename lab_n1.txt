#include <stdio.h>

// Функція приймає число d, а також вказівники на змінні S та D для запису результату
void sum_and_product_of_digits(int d, int *S, int *D) {
       // Виділяємо кожну цифру чотиризначного числа
        int thousands = d / 1000;                // Отримуємо тисячі
        int hundreds = (d / 100) % 10;           // Отримуємо сотні
        int tens = (d / 10) % 10;                // Отримуємо десятки
        int ones = d % 10;                       // Отримуємо одиниці
      
       // Обчислюємо суму (*S) та добуток (*D) за адресами в пам'яті
       *S = thousands + hundreds + tens + ones;
       *D = thousands * hundreds * tens * ones;
}

int main() {
     // Вхідні дані
     int d = 1352;
     int S = 0;
     int D = 0;

     // Виклик функції (передаємо адреси змінних S та D за допомогою &)
     sum_and_product_of_digits(d, &S, &D);

     // Виведення результату
     printf("Вхідні дані: d = %d\n", d);
     printf("Вихідні дані: S = %d, D = %d\n", S, D);

     return 0;
}
